43   /**
44    * Load data from a file and return the object depending on which [[JsonParser]] type is passed
45    *
46    * @param jsonParser
47    * @param filePath
48    * @tparam A
49    * @return a Try[A]
50    */
51   def loadDataFromFile[A](jsonParser: JsonParser[A], filePath: String): Try[A] =
52     FileHandler.readFile(filePath) match
53       case Success(jsonString: String) =>
54         for {
55           jsonParser <- jsonParser.deserialize(jsonString)
56         } yield {
57           jsonParser match
58             case savedCourseList: List[SavedCourse] => savedCourseList
59             case playerStats: PlayerStats => playerStats
60         }
61       case Failure(f) => Failure(f)

67   /**
68    * Merge the elements of two list of [[SavedCourses]] using [[List.groupMapReduce]].
69    * If two or more elements have the same id, it checks if their value are the same,
70    *
71    * @param savedCourseList
72    * @param newSavedCourseList
73    * @return a resulting List[SavedCourse]
74    */
75   def mergeSavedCourseLists(savedCourseList: List[SavedCourse], newSavedCourseList: List[SavedCourse]): List[SavedCourse] =
76     (savedCourseList ::: newSavedCourseList).groupMapReduce(sc => sc.courseId)(sc => (sc.description, sc.quizList))(
77       (savedCourse, newSavedCourse) => (savedCourse._1, mergeQuizLists(savedCourse._2, newSavedCourse._2))
78     ).map((k, v) => SavedCourse(k, v._1, v._2)).toList
79 
80   // Used by mergeSavedCourseLists to merge quiz lists
81   private def mergeQuizLists(quizList: List[Quiz], newQuizList: List[Quiz]): List[Quiz] =
82     (quizList ::: newQuizList).groupMapReduce(q => q.quizId)(q => (q.question, q.answerList, q.maxScore, q.imagePath))(
83       (quiz, newQuiz) => (quiz._1, mergeAnswerLists(quiz._2, newQuiz._2), quiz._3, quiz._4)
84     ).map((k, v) => Quiz(k, v._1, v._2, v._3, v._4)).toList
85 
86   // Used by mergeQuizLists to merge answer lists
87   private def mergeAnswerLists(answerList: List[Answer], newAnswerList: List[Answer]): List[Answer] =
88     (answerList ::: newAnswerList).groupMapReduce(a => a.text)(a => a.isCorrect)(
89       (answer, newAnswer) => answer
90     ).map((k, v) => Answer(k, v)).toList
