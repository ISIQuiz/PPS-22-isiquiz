26   /**

28    *
29    * @param data
30    * @param filePath
31    * @tparam A
32    * @return a Try[String] with file path
33    */
34   def storeDataToPath[A](data: A, filePath: String): Try[String] =
35     FileHandler.createDirectory(PlayerDataDirectoryPath)

44   /**
45    * Load data from a file and return the object depending on which [[JsonParser]] type is passed
46    *
47    * @param jsonParser
48    * @param filePath
49    * @tparam A
50    * @return a Try[A]
51    */
52   def loadDataFromFile[A](jsonParser: JsonParser[A], filePath: String): Try[A] =
53     FileHandler.readFile(filePath) match
54       case Success(jsonString: String) =>
55         for {
56           jsonParser <- jsonParser.deserialize(jsonString)
57         } yield {
58           jsonParser match
59             case savedCourseList: List[SavedCourse] => savedCourseList
60             case playerStats: PlayerStats => playerStats
61         }
62       case Failure(f) => Failure(f)

68   /**
69    * Merge the elements of two list of [[SavedCourses]] using [[List.groupMapReduce]].
70    * If two or more elements have the same id, it checks if their value are the same,
71    *
72    * @param savedCourseList
73    * @param newSavedCourseList
74    * @return a resulting List[SavedCourse]
75    */
76   def mergeSavedCourseLists(savedCourseList: List[SavedCourse], newSavedCourseList: List[SavedCourse]): List[SavedCourse] =
77     (savedCourseList ::: newSavedCourseList).groupMapReduce(sc => sc.courseId)(sc => (sc.description, sc.quizList))(
78       (savedCourse, newSavedCourse) => (savedCourse._1, mergeQuizLists(savedCourse._2, newSavedCourse._2))
79     ).map((k, v) => SavedCourse(k, v._1, v._2)).toList
80 
81   // Used by mergeSavedCourseLists to merge quiz lists
82   private def mergeQuizLists(quizList: List[Quiz], newQuizList: List[Quiz]): List[Quiz] =
83     (quizList ::: newQuizList).groupMapReduce(q => q.quizId)(q => (q.question, q.answerList, q.maxScore, q.imagePath))(
84       (quiz, newQuiz) => (quiz._1, mergeAnswerLists(quiz._2, newQuiz._2), quiz._3, quiz._4)
85     ).map((k, v) => Quiz(k, v._1, v._2, v._3, v._4)).toList
86 
87   // Used by mergeQuizLists to merge answer lists
88   private def mergeAnswerLists(answerList: List[Answer], newAnswerList: List[Answer]): List[Answer] =
89     (answerList ::: newAnswerList).groupMapReduce(a => a.text)(a => a.isCorrect)(
90       (answer, newAnswer) => answer
91     ).map((k, v) => Answer(k, v)).toList
