57   /**
58    * Merge the second object into the first object. It's similar to JSON play library [[JsObject.deepMerge]] method.
59    * The key difference is that can merge nested JsArrays if starting JsObjects have a similar structure
60    *
61    * @param firstObject  first JsObject to be merged
62    * @param secondObject second JsObject to be merged
63    * @return a merged JsObject
64    */
65   def merge(firstObject: JsObject, secondObject: JsObject): JsObject =
66     val result = firstObject.fields ++ secondObject.fields.map {
67       case (secondKey, secondValue) =>
68         val maybeExistingValue = firstObject.value.get(secondKey)
69         // if exist two value with same key do merge
70         val newValue = (maybeExistingValue, secondValue) match {
71           case (Some(f: JsObject), s: JsObject) => merge(f, s) // merge JsObjects
72           case (Some(f: JsArray), s: JsArray) => JsArray(f.value.zip(s.value).map { // merge every JsObjects of JsArray
73             case (f: JsObject, s: JsObject) => merge(f, s)
74             case _ => JsObject.empty
75           })
76           case _ => secondValue
77         }
78         secondKey -> newValue
79     }
80     JsObject(result)
