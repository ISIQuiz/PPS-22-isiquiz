64   def selectAnswer[T](actionParameter: Option[T]): Unit =
65     timer.stopTimer()

67       val selectedAnswerIndex: Int = actionParameter.get.asInstanceOf[Int]

102   def updateScore(timeRemaining: Int, maxTime: Long, maxScore: Int): Int = {
103     val coeff: Double = maxScore.toDouble / (maxTime / 1000).toDouble
104     val score = (coeff * timeRemaining).toInt

108   }
