68   def selectAnswer[T](actionParameter: Option[T]): Unit =
69     currentTime = timer.getTime
70     timer.stopTimer()
71     if actionParameter.isDefined && !timer.isExpired then
72       actionParameter match

80   override def nextQuiz(): QuizInGame =
81     val maxTime = timer.maxTime/1000
82     currentAnswer match
83       case Some(ans) =>
84         if (!ans.isCorrect)
85           currentScore = 0
86           currentTime = maxTime
87         gameStage.addReviewQuizAnswer(ans, currentScore, currentTime)
88         gameStage.addQuizToStats(ans.isCorrect, currentScore, currentTime)
89 
90       case _ =>
91         gameStage.addReviewQuizNotAnswered()
92         gameStage.addQuizToStats(false, 0, maxTime)
93 
94     if gameStage.maxQuizzesReached then endGame()
95     currentAnswer = None
96     currentScore = 0
97     currentTime = 0
98     timer.startTimer()
99     extractQuizInGame()

112   override def chooseAnswers(quiz: Quiz): List[Answer] =
113     val allCorrectAnswers = quiz.answerList.filter(answer => answer.isCorrect)
114     val allWrongAnswers = quiz.answerList.filter(answer => !answer.isCorrect)
115 
116     val correctAnswers = randomNumberGenerator(1, allCorrectAnswers.size).map(allCorrectAnswers)
117     val wrongAnswers = randomNumberGenerator(3, allWrongAnswers.size).map(allWrongAnswers)
118 
119     scala.util.Random.shuffle(correctAnswers ::: wrongAnswers)

121   // update the game score based on time remaining
122   private def updateScore(timeRemaining: Int, maxTime: Long, maxScore: Int): Int = {
123     val coeff: Double = maxScore.toDouble / (maxTime / 1000).toDouble
124     val score = (coeff * timeRemaining).toInt
125     if (score >= maxScore) score else score + 1
126   }
