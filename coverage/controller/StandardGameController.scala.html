56   override def handle[T](action: Action[T]): Unit = action match

61   override def nextIteration(): Unit =
62     sendUpdate(DefaultUpdate)
63     if timer.isExpired then
64       timer.stopTimer()
65       sendUpdate(TimeExpiredUpdate)
66     else
67       sendUpdate(CurrentGameUpdate(Option(gameStage)))
68       if !timer.isStopped then
69         sendUpdate(TimerUpdate(Option(timer)))
70         // Update actual quiz in game score
71         currentScore = updateScore(timer.getRemainingTime.toInt, timer.maxTimeInSeconds, gameStage.quizInGame.quiz.maxScore)
72         sendUpdate(QuizScoreUpdate(Option(currentScore)))

75   def selectAnswer[T](actionParameter: Option[T]): Unit =
76     currentTime = timer.getTime
77     timer.stopTimer()
78     if actionParameter.isDefined && !timer.isExpired then
79       actionParameter match
80         case Some(selectedAnswerIndex: Int) =>
81           currentAnswer = Some(gameStage.quizInGame.answers(selectedAnswerIndex))
82         case Some(selectedAnswer: Answer) =>
83           currentAnswer = Some(selectedAnswer)
84         case _ => throw IllegalArgumentException()
85       sendUpdate(AnswerFeedbackUpdate(Option((gameStage.quizInGame.answers.indexOf(currentAnswer.get), currentAnswer.get.isCorrect))))
86 
87   override def nextQuiz(): Unit =
88     currentAnswer match
89       case Some(ans) =>
90         gameStage.addReviewQuizAnswer(ans, currentScore, currentTime)
91         gameStage.addQuizToStats(ans.isCorrect, currentScore, currentTime)
92       case _ =>
93         gameStage.addReviewQuizNotAnswered()
94         gameStage.addQuizToStats(false, 0, 0)
95 
96     if gameStage.maxQuizzesReached then endGame(gameStage)
97     currentAnswer = None
98     currentScore = 0
99     currentTime = 0
100     timer.startTimer()
101     gameStage.quizInGame = extractQuizInGame(gameStage)
102 
103   // update the game score based on time remaining
104   private def updateScore(timeRemaining: Int, maxTimeInSeconds: Long, maxScore: Int): Int = {
105     val coeff: Double = maxScore.toDouble / maxTimeInSeconds.toDouble
106     val score = (coeff * timeRemaining).toInt
107     if (score >= maxScore) score else score + 1
108   }
