56   override def handle[T](action: Action[T]): Unit = action match

61   override def nextIteration(): Unit =
62     sendUpdate(DefaultUpdate)
63     if timer.isExpired then
64       timer.stopTimer()
65       sendUpdate(TimeExpiredUpdate)
66     else
67       sendUpdate(CurrentGameUpdate(Option(gameStage)))
68       if !timer.isStopped then
69         sendUpdate(TimerUpdate(Option(timer)))
70         // Update actual quiz in game score
71         currentScore = updateScore(timer.getRemainingTime.toInt, timer.maxTimeInSeconds, gameStage.quizInGame.quiz.maxScore)
72         sendUpdate(QuizScoreUpdate(Option(currentScore)))
73 
74   def selectAnswer[T](actionParameter: Option[T]): Unit =
75     currentTime = timer.getTime
76     timer.stopTimer()
77     if actionParameter.isDefined && !timer.isExpired then
78       actionParameter match
79         case Some(selectedAnswerIndex: Int) =>
80           currentAnswer = Some(gameStage.quizInGame.answers(selectedAnswerIndex))
81         case Some(selectedAnswer: Answer) =>
82           currentAnswer = Some(selectedAnswer)
83         case _ => throw IllegalArgumentException()
84       sendUpdate(AnswerFeedbackUpdate(Option((gameStage.quizInGame.answers.indexOf(currentAnswer.get), currentAnswer.get.isCorrect))))
85 
86   override def nextQuiz(): Unit =
87     currentAnswer match
88       case Some(ans) =>
89         gameStage.addReviewQuizAnswer(ans, currentScore, currentTime)
90         gameStage.addQuizToStats(ans.isCorrect, currentScore, currentTime)
91       case _ =>
92         gameStage.addReviewQuizNotAnswered()
93         gameStage.addQuizToStats(false, 0, 0)
94     if gameStage.maxQuizzesReached then endGame(gameStage)
95     currentAnswer = None
96     currentScore = 0
97     currentTime = 0
98     timer.startTimer()
99     gameStage.quizInGame = extractQuizInGame(gameStage)
100 
101   // update the game score based on time remaining
102   private def updateScore(timeRemaining: Int, maxTimeInSeconds: Long, maxScore: Int): Int = {
103     val coeff: Double = maxScore.toDouble / maxTimeInSeconds.toDouble
104     val score = (coeff * timeRemaining).toInt
105     if (score >= maxScore) score else score + 1
106   }
