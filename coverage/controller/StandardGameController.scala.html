64   def selectAnswer[T](actionParameter: Option[T]): Unit =
65     timer.stopTimer()
66     if actionParameter.isDefined && !timer.isExpired then
67       actionParameter match
68         case Some(selectedAnswerIndex:Int) =>
69           currentAnswer = Some(gameStage.quizInGame.answers(selectedAnswerIndex))
70         case Some(selectedAnswer: Answer) =>
71           currentAnswer = Some(selectedAnswer)

96   override def chooseAnswers(quiz: Quiz): List[Answer] =
97     val allCorrectAnswers = quiz.answerList.filter(answer => answer.isCorrect)
98     val allWrongAnswers = quiz.answerList.filter(answer => !answer.isCorrect)
99 
100     val correctAnswers = randomNumberGenerator(1, allCorrectAnswers.size).map(allCorrectAnswers)
101     val wrongAnswers = randomNumberGenerator(3, allWrongAnswers.size).map(allWrongAnswers)
102 
103     scala.util.Random.shuffle(correctAnswers ::: wrongAnswers)

105   def updateScore(timeRemaining: Int, maxTime: Long, maxScore: Int): Int = {
106     val coeff: Double = maxScore.toDouble / (maxTime / 1000).toDouble
107     val score = (coeff * timeRemaining).toInt
108     if (score >= maxScore) score else score + 1
109   }
