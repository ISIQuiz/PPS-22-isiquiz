70   def selectAnswer[T](actionParameter: Option[T]): Unit =
71     currentTime = timer.getTime
72     timer.stopTimer()
73     if actionParameter.isDefined && !timer.isExpired then
74       actionParameter match
75         case Some(selectedAnswerIndex:Int) =>
76           currentAnswer = Some(gameStage.quizInGame.answers(selectedAnswerIndex))
77         case Some(selectedAnswer: Answer) =>
78           currentAnswer = Some(selectedAnswer)
79         case _ => throw IllegalArgumentException()
80       sendUpdate(AnswerFeedbackUpdate(Option((gameStage.quizInGame.answers.indexOf(currentAnswer.get), currentAnswer.get.isCorrect))))

82   override def nextQuiz(): QuizInGame =
83     val maxTime = timer.maxTime/1000
84     currentAnswer match
85       case Some(ans) =>
86         if (!ans.isCorrect)
87           currentScore = 0
88           currentTime = maxTime
89         gameStage.addReviewQuizAnswer(ans, currentScore, currentTime)
90         gameStage.addQuizToStats(ans.isCorrect, currentScore, currentTime)
91 
92       case _ =>
93         gameStage.addReviewQuizNotAnswered()
94         gameStage.addQuizToStats(false, 0, maxTime)
95 
96     if gameStage.maxQuizzesReached then endGame()
97     currentAnswer = None
98     currentScore = 0
99     currentTime = 0
100     timer.startTimer()
101     extractQuizInGame()

104   def extractQuizInGame(): QuizInGame =
105     val selectedCourse = gameStage.coursesInGame(randomNumberGenerator(1, gameStage.coursesInGame.size).head)
106     val selectedQuiz = chooseQuiz(selectedCourse)
107     val selectedAnswers = chooseAnswers(selectedQuiz)
108     val quizInGame = QuizInGame(selectedCourse, selectedQuiz, selectedAnswers)
109     gameStage.quizInGame_(quizInGame)
110     quizInGame

112   override def chooseQuiz(course: SavedCourse): Quiz = course.quizList(randomNumberGenerator(1, course.quizList.size).head)

114   override def chooseAnswers(quiz: Quiz): List[Answer] =
115     val allCorrectAnswers = quiz.answerList.filter(answer => answer.isCorrect)
116     val allWrongAnswers = quiz.answerList.filter(answer => !answer.isCorrect)
117 
118     val correctAnswers = randomNumberGenerator(1, allCorrectAnswers.size).map(allCorrectAnswers)
119     val wrongAnswers = randomNumberGenerator(3, allWrongAnswers.size).map(allWrongAnswers)
120 
121     scala.util.Random.shuffle(correctAnswers ::: wrongAnswers)
122 
123   // update the game score based on time remaining
124   private def updateScore(timeRemaining: Int, maxTime: Long, maxScore: Int): Int = {
125     val coeff: Double = maxScore.toDouble / (maxTime / 1000).toDouble
126     val score = (coeff * timeRemaining).toInt
127     if (score >= maxScore) score else score + 1
128   }
