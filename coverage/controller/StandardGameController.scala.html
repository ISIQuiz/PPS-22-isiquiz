40 /** Defines the logic of the select page */
41 class StandardGameController(val game: GameStage) extends PageController, GameController:

51   override def handle[T](action: Action[T]): Unit = action match
52     case Back => AppController.handle(MainMenuAction)

56   override def nextIteration(): Unit =
57     sendUpdate(DefaultUpdate)
58     if timer.isExpired then
59       timer.stopTimer()
60       sendUpdate(TimeExpiredUpdate)
61     else
62       sendUpdate(CurrentGameUpdate(Option(gameStage)))
63       if !timer.isStopped then
64         sendUpdate(TimerUpdate(Option(timer)))
65         // Update actual quiz in game score
66         currentScore = updateScore(timer.getRemainingTime.toInt, timer.maxTime, gameStage.quizInGame.quiz.maxScore)
67         sendUpdate(QuizScoreUpdate(Option(currentScore)))

70   def selectAnswer[T](actionParameter: Option[T]): Unit =
71     currentTime = timer.getTime
72     timer.stopTimer()
73     if actionParameter.isDefined && !timer.isExpired then
74       actionParameter match
75         case Some(selectedAnswerIndex:Int) =>
76           currentAnswer = Some(gameStage.quizInGame.answers(selectedAnswerIndex))
77         case Some(selectedAnswer: Answer) =>
78           currentAnswer = Some(selectedAnswer)
79         case _ => throw IllegalArgumentException()
80       sendUpdate(AnswerFeedbackUpdate(Option((gameStage.quizInGame.answers.indexOf(currentAnswer.get), currentAnswer.get.isCorrect))))
81 
82   override def nextQuiz(): Unit =
83     val maxTime = timer.maxTime/1000
84     currentAnswer match
85       case Some(ans) =>
86         if (!ans.isCorrect)
87           currentScore = 0
88           currentTime = maxTime
89         gameStage.addReviewQuizAnswer(ans, currentScore, currentTime)
90         gameStage.addQuizToStats(ans.isCorrect, currentScore, currentTime)
91 
92       case _ =>
93         gameStage.addReviewQuizNotAnswered()
94         gameStage.addQuizToStats(false, 0, maxTime)
95 
96     if gameStage.maxQuizzesReached then endGame(gameStage)
97     currentAnswer = None
98     currentScore = 0
99     currentTime = 0
100     timer.startTimer()
101     gameStage.quizInGame = extractQuizInGame(gameStage)
102 
103   // update the game score based on time remaining
104   private def updateScore(timeRemaining: Int, maxTime: Long, maxScore: Int): Int = {
105     val coeff: Double = maxScore.toDouble / (maxTime / 1000).toDouble
106     val score = (coeff * timeRemaining).toInt
107     if (score >= maxScore) score else score + 1
108   }
