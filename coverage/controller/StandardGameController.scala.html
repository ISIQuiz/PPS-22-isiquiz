54   override def nextIteration(): Unit =
55     sendUpdate(DefaultUpdate)
56     if timer.isExpired then
57       timer.stopTimer()
58       sendUpdate(TimeExpiredUpdate)
59     else
60       sendUpdate(CurrentGameUpdate(Option(gameStage)))

68   def selectAnswer[T](actionParameter: Option[T]): Unit =
69     currentTime = timer.getTime
70     timer.stopTimer()
71     if actionParameter.isDefined && !timer.isExpired then
72       val selectedAnswerIndex: Int = actionParameter.get.asInstanceOf[Int]
73       val answerSelected: Answer = gameStage.quizInGame.answers(selectedAnswerIndex)
74       currentAnswer = Some(answerSelected)
75       sendUpdate(AnswerFeedbackUpdate(Option((selectedAnswerIndex, answerSelected.isCorrect))))

77   override def nextQuiz(): QuizInGame =
78     val maxTime = timer.maxTime/1000
79     currentAnswer match
80       case Some(ans) =>
81         if (!ans.isCorrect)
82           currentScore = 0
83           currentTime = maxTime
84         gameStage.addReviewQuizAnswer(ans, currentScore, currentTime)
85         gameStage.addQuizToStats(ans.isCorrect, currentScore, currentTime)
86 
87       case _ =>
88         gameStage.addReviewQuizNotAnswered()
89         gameStage.addQuizToStats(false, 0, maxTime)
90 
91     if gameStage.maxQuizzesReached then endGame()
92     currentAnswer = None
93     currentScore = 0
94     currentTime = 0
95     timer.startTimer()
96     extractQuizInGame()

99   def extractQuizInGame(): QuizInGame =
100     val selectedCourse = gameStage.coursesInGame(randomNumberGenerator(1, gameStage.coursesInGame.size).head)
101     val selectedQuiz = chooseQuiz(selectedCourse)
102     val selectedAnswers = chooseAnswers(selectedQuiz)

104     gameStage.quizInGame_(quizInGame)

107   override def chooseQuiz(course: SavedCourse): Quiz = course.quizList(randomNumberGenerator(1, course.quizList.size).head)

109   override def chooseAnswers(quiz: Quiz): List[Answer] =
110     val allCorrectAnswers = quiz.answerList.filter(answer => answer.isCorrect)
111     val allWrongAnswers = quiz.answerList.filter(answer => !answer.isCorrect)
112 
113     val correctAnswers = randomNumberGenerator(1, allCorrectAnswers.size).map(allCorrectAnswers)
114     val wrongAnswers = randomNumberGenerator(3, allWrongAnswers.size).map(allWrongAnswers)
115 
116     scala.util.Random.shuffle(correctAnswers ::: wrongAnswers)
117 
118   // update the game score based on time remaining
119   private def updateScore(timeRemaining: Int, maxTime: Long, maxScore: Int): Int = {
120     val coeff: Double = maxScore.toDouble / (maxTime / 1000).toDouble
121     val score = (coeff * timeRemaining).toInt
122     if (score >= maxScore) score else score + 1
123   }
