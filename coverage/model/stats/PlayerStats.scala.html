116   def addQuizInGameToStats(playerStats: PlayerStats, quizInGame: QuizInGame, isCorrect: Boolean, score: Int, timeToAnswer: Double): PlayerStats =
117     val newQuizInStats = QuizInStats(quizInGame.quiz.quizId, 1, score, if (isCorrect) 1 else 0, timeToAnswer)
118     val newCourseInStats = CourseInStats(quizInGame.course, List(newQuizInStats))
119 
120     // Check if course already exists
121     val newCourseInStatsList = if (playerStats.courseInStatsList.map(c => c.course).contains(quizInGame.course))
122     // Merge existent player stats with the new player stats
123       mergeCourseInStatsLists(playerStats.courseInStatsList, List(newCourseInStats))
124     else
125       playerStats.courseInStatsList.appended(newCourseInStats)
126 
127     updatePlayerStats(newCourseInStatsList)

140   private def mergeCourseInStatsLists(courseInStatsList: List[CourseInStats], newCourseInStatsList: List[CourseInStats]): List[CourseInStats] =
141     (courseInStatsList ::: newCourseInStatsList).groupMapReduce(c => c.course)(c => (c.quizInStatsList))(
142       (courseInStats, newCourseInStats) => mergeQuizInStatsLists(courseInStats, newCourseInStats)

145   // Merge two lists of QuizInStats and calculates the new values
146   private def mergeQuizInStatsLists(quizInStatsList: List[QuizInStats], newQuizInStatsList: List[QuizInStats]): List[QuizInStats] =
147     (quizInStatsList ::: newQuizInStatsList).groupMapReduce(q => q.quizId)(q => (q.totalSeen, q.totalScore, q.totalRightAnswers, q.averageTimeAnswer))(
148       (quizInStats, newQuizInStats) => (
149         quizInStats._1 + newQuizInStats._1, // sum total seen
150         quizInStats._2 + newQuizInStats._2, // sum total score
151         quizInStats._3 + newQuizInStats._3, // sum total right answer
152         calculateWeightedAverage(quizInStats._4, quizInStats._1, newQuizInStats._4, newQuizInStats._1) //calculate weighted average
153       )
154     ).map((k, v) => QuizInStats(k, v._1, v._2, v._3, v._4)).toList

156   // Calculate weighted average
157   private def calculateWeightedAverage(averageTimeAnswer: Double, totalSeen: Int, newAverageTimeAnswer: Double, newTotalSeen: Int): Double =
158     val c: Double = Try(
159       ((averageTimeAnswer * totalSeen) + (newAverageTimeAnswer * newTotalSeen)) / (totalSeen + newTotalSeen)
160     ).getOrElse(0)
161     BigDecimal(c).setScale(2, BigDecimal.RoundingMode.HALF_UP).toDouble
