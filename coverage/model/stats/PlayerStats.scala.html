116   def addQuizInGameToStats(playerStats: PlayerStats, quizInGame: QuizInGame, isCorrect: Boolean, score: Int, timeToAnswer: Double): PlayerStats =
117 
118     val newQuizInStats = if (isCorrect)
119       QuizInStats(quizInGame.quiz.quizId, 1, score, 1, timeToAnswer)
120     else
121       QuizInStats(quizInGame.quiz.quizId, 1)

123     val newCourseInStats = CourseInStats(quizInGame.course, List(newQuizInStats))
124 
125     // Check if course already exists
126     val newCourseInStatsList = if (playerStats.courseInStatsList.map(c => c.course).contains(quizInGame.course))
127     // Merge existent player stats with the new player stats
128       mergeCourseInStatsLists(playerStats.courseInStatsList, List(newCourseInStats))
129     else
130       playerStats.courseInStatsList.appended(newCourseInStats)

146     (courseInStatsList ::: newCourseInStatsList).groupMapReduce(c => c.course.courseId)(c => c.quizInStatsList)(

150   // Merge two lists of QuizInStats and calculates the new values
151   private def mergeQuizInStatsLists(quizInStatsList: List[QuizInStats], newQuizInStatsList: List[QuizInStats]): List[QuizInStats] =
152     (quizInStatsList ::: newQuizInStatsList).groupMapReduce(q => q.quizId)(q => (q.totalSeen, q.totalScore, q.totalRightAnswers, q.averageTimeAnswer))(
153       (quizInStats, newQuizInStats) => (
154         quizInStats._1 + newQuizInStats._1, // sum total seen
155         quizInStats._2 + newQuizInStats._2, // sum total score
156         quizInStats._3 + newQuizInStats._3, // sum total right answer
157         calculateWeightedAverage(quizInStats._4, quizInStats._3, newQuizInStats._4, newQuizInStats._3) //calculate weighted average
158       )
159     ).map((k, v) => QuizInStats(k, v._1, v._2, v._3, v._4)).toList

161   // Calculate weighted average time to answer only on correct answer
162   private def calculateWeightedAverage(averageTimeAnswer: Double, totalCorrectAnswers: Int, newAverageTimeAnswer: Double, newTotalCorrectAnswers: Int): Double =
163     val c: Double = if ((totalCorrectAnswers + newTotalCorrectAnswers) > 0)
164       ((averageTimeAnswer * totalCorrectAnswers) + (newAverageTimeAnswer * newTotalCorrectAnswers)) / (totalCorrectAnswers + newTotalCorrectAnswers)
165     else
166       0
167     BigDecimal(c).setScale(2, BigDecimal.RoundingMode.HALF_UP).toDouble

197   // Calculates total average time to answer
198   private def calculateTotalAverageTimeAnswer(courseInStatsList: List[CourseInStats]): Double =
199     val totalCorrectAnswer = calculateTotalCorrectAnswer(courseInStatsList)
200     if (courseInStatsList.nonEmpty && totalCorrectAnswer != 0)
201       val totalAverageTimeAnswer = courseInStatsList.flatMap(c => c.quizInStatsList).map(q => q.averageTimeAnswer * q.totalRightAnswers).sum
202       val avg = totalAverageTimeAnswer / totalCorrectAnswer
203       BigDecimal(avg).setScale(2, BigDecimal.RoundingMode.HALF_UP).toDouble
204     else
