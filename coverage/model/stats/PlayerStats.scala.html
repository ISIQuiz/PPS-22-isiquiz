125    *
126    * @param playerStats

129    * @param score

133   def addQuizInGameToStats(playerStats: PlayerStats, quizInGame: QuizInGame, isCorrect: Boolean, score: Int, timeToAnswer: Double): PlayerStats =
134     val newQuizInStats = QuizInStats(quizInGame.quiz.quizId, 1, score, if (isCorrect) 1 else 0, timeToAnswer)
135     val newCourseInStats = CourseInStats(quizInGame.course, List(newQuizInStats))
136 
137     // Check if course already exists
138     val newCourseInStatsList = if (playerStats.courseInStatsList.map(c => c.course).contains(quizInGame.course))
139     // Merge existent player stats with the new player stats
140       mergeCourseInStatsLists(playerStats.courseInStatsList, List(newCourseInStats))
141     else
142       playerStats.courseInStatsList.appended(newCourseInStats)
143 
144     updatePlayerStats(newCourseInStatsList)

158   private def mergeCourseInStatsLists(courseInStatsList: List[CourseInStats], newCourseInStatsList: List[CourseInStats]): List[CourseInStats] =
159     (courseInStatsList ::: newCourseInStatsList).groupMapReduce(c => c.course)(c => (c.quizInStatsList))(
160       (courseInStats, newCourseInStats) => mergeQuizInStatsLists(courseInStats, newCourseInStats)
161     ).map((k, v) => CourseInStats(k, v)).toList

163   // Merge two lists of QuizInStats and calculates the new values
164   private def mergeQuizInStatsLists(quizInStatsList: List[QuizInStats], newQuizInStatsList: List[QuizInStats]): List[QuizInStats] =
165     (quizInStatsList ::: newQuizInStatsList).groupMapReduce(q => q.quizId)(q => (q.totalSeen, q.totalScore, q.totalRightAnswers, q.averageTimeAnswer))(
166       (quizInStats, newQuizInStats) => (
167         quizInStats._1 + newQuizInStats._1, // sum total seen
168         quizInStats._2 + newQuizInStats._2, // sum total score
169         quizInStats._3 + newQuizInStats._3, // sum total right answer
170         calculateWeightedAverage(quizInStats._4, quizInStats._1, newQuizInStats._4, newQuizInStats._1) //calculate weighted average
171       )
172     ).map((k, v) => QuizInStats(k, v._1, v._2, v._3, v._4)).toList

174   // Needed to calculate weighted average
175   private def calculateWeightedAverage(averageTimeAnswer: Double, totalSeen: Int, newAverageTimeAnswer: Double, newTotalSeen: Int): Double =
176     Try(
177       ((averageTimeAnswer * (if (totalSeen == 0) 1 else totalSeen)) + newAverageTimeAnswer) / (totalSeen + newTotalSeen)
178     ).getOrElse(averageTimeAnswer)
