118   def addQuizInGameToStats(playerStats: PlayerStats, quizInGame: QuizInGame, isCorrect: Boolean, score: Int, timeToAnswer: Double): PlayerStats =
119 
120     val newQuizInStats = if (isCorrect)
121       QuizInStats(quizInGame.quiz.quizId, 1, score, 1, timeToAnswer)

123       QuizInStats(quizInGame.quiz.quizId, 1)

126 
127     // Check if course already exists

129     // Merge existent player stats with the new player stats
130       mergeCourseInStatsLists(playerStats.courseInStatsList, List(newCourseInStats))
131     else
132       playerStats.courseInStatsList.appended(newCourseInStats)

148     (courseInStatsList ::: newCourseInStatsList).groupMapReduce(c => c.course.courseId)(c => c.quizInStatsList)(

152   // Merge two lists of QuizInStats and calculates the new values
153   private def mergeQuizInStatsLists(quizInStatsList: List[QuizInStats], newQuizInStatsList: List[QuizInStats]): List[QuizInStats] =
154     (quizInStatsList ::: newQuizInStatsList).groupMapReduce(q => q.quizId)(q => (q.totalSeen, q.totalScore, q.totalRightAnswers, q.averageTimeAnswer))(
155       (quizInStats, newQuizInStats) => (
156         quizInStats._1 + newQuizInStats._1, // sum total seen
157         quizInStats._2 + newQuizInStats._2, // sum total score
158         quizInStats._3 + newQuizInStats._3, // sum total right answer
159         calculateWeightedAverage(quizInStats._4, quizInStats._3, newQuizInStats._4, newQuizInStats._3) //calculate weighted average
160       )
161     ).map((k, v) => QuizInStats(k, v._1, v._2, v._3, v._4)).toList

163   // Calculate weighted average time to answer only on correct answer
164   private def calculateWeightedAverage(averageTimeAnswer: Double, totalCorrectAnswers: Int, newAverageTimeAnswer: Double, newTotalCorrectAnswers: Int): Double =
165     val c: Double = if ((totalCorrectAnswers + newTotalCorrectAnswers) > 0)
166       ((averageTimeAnswer * totalCorrectAnswers) + (newAverageTimeAnswer * newTotalCorrectAnswers)) / (totalCorrectAnswers + newTotalCorrectAnswers)
167     else
168       0
169     BigDecimal(c).setScale(2, BigDecimal.RoundingMode.HALF_UP).toDouble

200   // Calculates total average time to answer
201   private def calculateTotalAverageTimeAnswer(courseInStatsList: List[CourseInStats]): Double =
202     val totalCorrectAnswer = calculateTotalCorrectAnswer(courseInStatsList)
203     if (courseInStatsList.nonEmpty && totalCorrectAnswer != 0)
204       val totalAverageTimeAnswer = courseInStatsList.flatMap(c => c.quizInStatsList).map(q => q.averageTimeAnswer * q.totalRightAnswers).sum
205       val avg = totalAverageTimeAnswer / totalCorrectAnswer
206       BigDecimal(avg).setScale(2, BigDecimal.RoundingMode.HALF_UP).toDouble
207     else
