167           coursesVBox.getChildren.addAll(radioButton)

169       }
170     case QuizListUpdate(updateParameter) =>
171       Platform.runLater { () =>
172         quizVBox.getChildren.clear()

174           val radioButton = RadioButton(quiz.question);

178             radioButton.setSelected(true);

181             scoreIntegerField.setText(quiz.maxScore.toString);
182             clearAllAnswers();
183             quiz.answerList.foreach(ans => addAnswerGUI(Option(ans)));
184             import controller.EditQuizMenuController.SelectQuizAction
185             sendEvent(SelectQuizAction(Option(quiz)));
186           );
187           quizVBox.getChildren.addAll(radioButton)
188         )
189       }

192       clearAllFields()
193     case QuizDeletedUpdate =>
