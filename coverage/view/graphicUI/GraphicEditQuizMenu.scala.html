147   override def updateUI[T](update: ViewUpdate[Any]): Unit = update match

165     case QuizListUpdate(updateParameter) =>
166       Platform.runLater { () =>

173             radioButton.setSelected(true);

179             import controller.EditQuizMenuController.SelectQuizAction
180             sendEvent(SelectQuizAction(Option(quiz)));
181           );
182           quizVBox.getChildren.addAll(radioButton)
183         )

187       clearAllFields()
