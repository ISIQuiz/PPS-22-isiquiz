65   @FXML
66   def importButtonClicked(): Unit =
67     val extensionFilter: ExtensionFilter = ExtensionFilter("JSON file", "*.json")
68     val fileChooser: FileChooser = FileChooser()
69     fileChooser.getExtensionFilters.add(extensionFilter)
70     fileChooser.setTitle(Vocabulary.IMPORT_QUIZ)
71     fileChooser.setInitialDirectory(File(CurrentDirectoryPath))
72     val selectedFile = fileChooser.showOpenDialog(stage)
73     if (selectedFile != null)
74       ImportHandler.importSavedCourseListFromFile(AppController.session, selectedFile.getPath) match
75         case Success(savedCourseList) =>
76           AppController.changeSavedCourses(savedCourseList)
77           ExportHandler.exportDataToPersonalDirectory(savedCourseList)
78           showDialogAlert(
79             AlertType.INFORMATION,
80             Vocabulary.IMPORT_QUIZ,
81             Vocabulary.OPERATION_SUCCEEDED,
82             selectedFile.getPath
83           )
84         case Failure(f) =>
85           showDialogAlert(
86             AlertType.ERROR,
87             Vocabulary.IMPORT_QUIZ,
