91   override def updateUI[T](update: ViewUpdate[Any]): Unit = update match
92     case DefaultUpdate => {}
93     case CurrentGameUpdate(updateParameter) =>
94       Platform.runLater(() => {
95         val gameStage: GameStage = updateParameter.get
96         courseLabel.setText(gameStage.quizInGame.course.courseId.courseName)
97         quizNumberLabel.setText(s"${gameStage.currentQuizNumber}/${gameStage.gameSettings.asInstanceOf[StandardGameSettings].maxQuizzes}")
98         quizLabel.setText(gameStage.quizInGame.quiz.question)
99         firstAnswerButton.setText(gameStage.quizInGame.answers(0).text)
100         secondAnswerButton.setText(gameStage.quizInGame.answers(1).text)
101         thirdAnswerButton.setText(gameStage.quizInGame.answers(2).text)
102         fourthAnswerButton.setText(gameStage.quizInGame.answers(3).text)
103       })
104     case AnswerFeedbackUpdate(updateParameter) =>
105       val feedback = updateParameter.get
106       feedback._1 match
107         case 0 => if feedback._2 then firstAnswerButton.getStyleClass.add("correct-answer") else firstAnswerButton.getStyleClass.add("wrong-answer")
108         case 1 => if feedback._2 then secondAnswerButton.getStyleClass.add("correct-answer") else secondAnswerButton.getStyleClass.add("wrong-answer")
109         case 2 => if feedback._2 then thirdAnswerButton.getStyleClass.add("correct-answer") else thirdAnswerButton.getStyleClass.add("wrong-answer")
110         case 3 => if feedback._2 then fourthAnswerButton.getStyleClass.add("correct-answer") else fourthAnswerButton.getStyleClass.add("wrong-answer")
111     case TimerUpdate(updateParameter) =>
112       val timer: Timer = updateParameter.get
113       Platform.runLater(() => {
114         timeRemainingLabel.setText(s"${timer.getRemainingTime.toInt}")
115         timeProgressBar.setProgress(timer.getCompletionPercentage)
116       })
117     case QuizScoreUpdate(updateParameter) =>
118       Platform.runLater(() => {
119         val score: Int = updateParameter.get
120         pointsLabel.setText(score + " punti")
121       })
122 
123     case TimeExpiredUpdate =>
124       disableButton(answerButtons)
125
