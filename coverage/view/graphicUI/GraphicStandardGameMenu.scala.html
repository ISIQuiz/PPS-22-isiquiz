91   override def updateUI[T](update: ViewUpdate[Any]): Unit = update match
92     case DefaultUpdate => {}
93     case CurrentGameUpdate(updateParameter) =>
94       Platform.runLater(() => {
95         val gameStage: GameStage = updateParameter.get
96         //maxScore = gameStage.quizInGame.quiz.maxScore
97         courseLabel.setText(gameStage.quizInGame.course.courseId.courseName)
98         quizNumberLabel.setText(s"${gameStage.currentQuizNumber}/${gameStage.gameSettings.asInstanceOf[StandardGameSettings].maxQuizzes}")
99         //pointsLabel.setText(actualScore.toString + " punti")
100         quizLabel.setText(gameStage.quizInGame.quiz.question)
101         firstAnswerButton.setText(gameStage.quizInGame.answers(0).text)
102         secondAnswerButton.setText(gameStage.quizInGame.answers(1).text)
103         thirdAnswerButton.setText(gameStage.quizInGame.answers(2).text)
104         fourthAnswerButton.setText(gameStage.quizInGame.answers(3).text)
105       })
106     case AnswerFeedbackUpdate(updateParameter) =>
107       val feedback = updateParameter.get
108       feedback._1 match
109         case 0 => if feedback._2 then firstAnswerButton.getStyleClass.add("correct-answer") else firstAnswerButton.getStyleClass.add("wrong-answer")
110         case 1 => if feedback._2 then secondAnswerButton.getStyleClass.add("correct-answer") else secondAnswerButton.getStyleClass.add("wrong-answer")
111         case 2 => if feedback._2 then thirdAnswerButton.getStyleClass.add("correct-answer") else thirdAnswerButton.getStyleClass.add("wrong-answer")
112         case 3 => if feedback._2 then fourthAnswerButton.getStyleClass.add("correct-answer") else fourthAnswerButton.getStyleClass.add("wrong-answer")
113     case TimerUpdate(updateParameter) =>
114       val timer: Timer = updateParameter.get
115       Platform.runLater(() => {
116         timeRemainingLabel.setText(s"${timer.getRemainingTime.toInt}")
117         timeProgressBar.setProgress(timer.getCompletionPercentage)
118         /*actualScore = updateScore(
119           timer.getRemainingTime.toInt,
120           timer.maxTime,
121           maxScore
122         )*/
123         //pointsLabel.setText(actualScore + " punti")
124       })
125     case QuizScoreUpdate(updateParameter) =>
126       Platform.runLater(() => {
127         val score: Int = updateParameter.get
128         pointsLabel.setText(score + " punti")
129       })
130 
131     case TimeExpiredUpdate =>
132       disableButton(answerButtons)
