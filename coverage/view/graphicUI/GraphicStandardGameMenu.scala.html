93   override def updateUI[T](update: ViewUpdate[Any]): Unit = update match
94     case DefaultUpdate => {}
95     case NewQuizUpdate(updateParameter) =>
96       Platform.runLater(() => {
97         val gameStage: GameStage = updateParameter.get.asInstanceOf[GameStage]
98         maxScore = gameStage.quizInGame.quiz.maxScore
99         courseLabel.setText(gameStage.quizInGame.course.courseId.courseName)
100         quizNumberLabel.setText(s"x/${gameStage.gameSettings.asInstanceOf[StandardGameSettings].maxQuizzes}")
101         pointsLabel.setText(maxScore.toString + " punti")
102         quizLabel.setText(gameStage.quizInGame.quiz.question)
103         firstAnswerButton.setText(gameStage.quizInGame.answers(0).text)
104         secondAnswerButton.setText(gameStage.quizInGame.answers(1).text)
105         thirdAnswerButton.setText(gameStage.quizInGame.answers(2).text)
106         fourthAnswerButton.setText(gameStage.quizInGame.answers(3).text)
107       })
108     case AnswerFeedbackUpdate(updateParameter) =>
109       val feedback = updateParameter.get.asInstanceOf[(Int, Boolean)]
110       feedback._1 match
111         case 0 => if feedback._2 then firstAnswerButton.getStyleClass.add("correct-answer") else firstAnswerButton.getStyleClass.add("wrong-answer")
112         case 1 => if feedback._2 then secondAnswerButton.getStyleClass.add("correct-answer") else secondAnswerButton.getStyleClass.add("wrong-answer")
113         case 2 => if feedback._2 then thirdAnswerButton.getStyleClass.add("correct-answer") else thirdAnswerButton.getStyleClass.add("wrong-answer")
114         case 3 => if feedback._2 then fourthAnswerButton.getStyleClass.add("correct-answer") else fourthAnswerButton.getStyleClass.add("wrong-answer")
115     case TimerUpdate(updateParameter) =>
116       val timer: Timer = updateParameter.get.asInstanceOf[Timer]
117       Platform.runLater(() => {
118         timeRemainingLabel.setText(s"${timer.getRemainingTime.toInt}")
119         timeProgressBar.setProgress(timer.getCompletionPercentage)
120         val newPoints =
121           updatePoints(
122             timer.getRemainingTime.toInt,
123             timer.maxTime,
124             maxScore
125           )
126         pointsLabel.setText(newPoints + " punti")
127       })
128 
129     case TimeExpiredUpdate =>
130       disableButton(answerButtons)

132   def updatePoints(timeRemaining: Int, maxTime: Long, maxScore: Int): Int = {
133     val coeff: Double = maxScore.toDouble / (maxTime / 1000).toDouble

135   }
