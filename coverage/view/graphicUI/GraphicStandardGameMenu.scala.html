94   override def updateUI[T](update: ViewUpdate[Any]): Unit = update match
95     case DefaultUpdate => {}
96     case NewQuizUpdate(updateParameter) =>
97       Platform.runLater(() => {
98         val gameStage: GameStage = updateParameter.get.asInstanceOf[GameStage]
99         maxScore = gameStage.quizInGame.quiz.maxScore
100         courseLabel.setText(gameStage.quizInGame.course.courseId.courseName)
101         quizNumberLabel.setText(s"x/${gameStage.gameSettings.asInstanceOf[StandardGameSettings].maxQuizzes}")
102         pointsLabel.setText(actualScore.toString + " punti")
103         quizLabel.setText(gameStage.quizInGame.quiz.question)
104         firstAnswerButton.setText(gameStage.quizInGame.answers(0).text)
105         secondAnswerButton.setText(gameStage.quizInGame.answers(1).text)
106         thirdAnswerButton.setText(gameStage.quizInGame.answers(2).text)
107         fourthAnswerButton.setText(gameStage.quizInGame.answers(3).text)
108       })
109     case AnswerFeedbackUpdate(updateParameter) =>
110       val feedback = updateParameter.get.asInstanceOf[(Int, Boolean)]
111       feedback._1 match
112         case 0 => if feedback._2 then firstAnswerButton.getStyleClass.add("correct-answer") else firstAnswerButton.getStyleClass.add("wrong-answer")
113         case 1 => if feedback._2 then secondAnswerButton.getStyleClass.add("correct-answer") else secondAnswerButton.getStyleClass.add("wrong-answer")
114         case 2 => if feedback._2 then thirdAnswerButton.getStyleClass.add("correct-answer") else thirdAnswerButton.getStyleClass.add("wrong-answer")
115         case 3 => if feedback._2 then fourthAnswerButton.getStyleClass.add("correct-answer") else fourthAnswerButton.getStyleClass.add("wrong-answer")
116     case TimerUpdate(updateParameter) =>
117       val timer: Timer = updateParameter.get.asInstanceOf[Timer]
118       Platform.runLater(() => {
119         timeRemainingLabel.setText(s"${timer.getRemainingTime.toInt}")
120         timeProgressBar.setProgress(timer.getCompletionPercentage)
121         actualScore = updateScore(
122           timer.getRemainingTime.toInt,
123           timer.maxTime,
124           maxScore
125         )
126         pointsLabel.setText(actualScore + " punti")
127       })
128 
129     case TimeExpiredUpdate =>
130       disableButton(answerButtons)

132   def updateScore(timeRemaining: Int, maxTime: Long, maxScore: Int): Int = {
133     val coeff: Double = maxScore.toDouble / (maxTime / 1000).toDouble
134     val score = (coeff * timeRemaining).toInt
135     /*println("ms: " + maxScore + "  |  mt: " + maxTime + "  |  c: " + coeff + "  |  tr" + timeRemaining)
136     println(score)*/
137     if (score >= maxScore) score else score + 1
138   }
