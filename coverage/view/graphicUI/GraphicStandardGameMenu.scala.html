97   override def updateUI[T](update: ViewUpdate[Any]): Unit = update match
98     case DefaultUpdate => {}
99     case CurrentGameUpdate(updateParameter) =>
100       Platform.runLater(() => {
101         val gameStage: GameStage = updateParameter.get
102         courseLabel.setText(gameStage.quizInGame.course.courseId.courseName)
103         quizNumberLabel.setText(s"${gameStage.currentQuizNumber}/${gameStage.gameSettings.asInstanceOf[StandardGameSettings].maxQuizzes}")
104         quizLabel.setText(gameStage.quizInGame.quiz.question)
105         firstAnswerButton.setText(gameStage.quizInGame.answers(0).text)
106         secondAnswerButton.setText(gameStage.quizInGame.answers(1).text)
107         thirdAnswerButton.setText(gameStage.quizInGame.answers(2).text)
108         fourthAnswerButton.setText(gameStage.quizInGame.answers(3).text)
109       })
110     case AnswerFeedbackUpdate(updateParameter) =>
111       val feedback = updateParameter.get
112       feedback._1 match
113         case 0 => if feedback._2 then firstAnswerButton.getStyleClass.add("correct-answer") else firstAnswerButton.getStyleClass.add("wrong-answer")
114         case 1 => if feedback._2 then secondAnswerButton.getStyleClass.add("correct-answer") else secondAnswerButton.getStyleClass.add("wrong-answer")
115         case 2 => if feedback._2 then thirdAnswerButton.getStyleClass.add("correct-answer") else thirdAnswerButton.getStyleClass.add("wrong-answer")
116         case 3 => if feedback._2 then fourthAnswerButton.getStyleClass.add("correct-answer") else fourthAnswerButton.getStyleClass.add("wrong-answer")
117     case TimerUpdate(updateParameter) =>
118       val timer: Timer = updateParameter.get
119       Platform.runLater(() => {
120         timeRemainingLabel.setText(s"${timer.getRemainingTime.toInt}")
121         timeProgressBar.setProgress(timer.getCompletionPercentage)
122       })
123     case QuizScoreUpdate(updateParameter) =>
124       Platform.runLater(() => {
125         val score: Int = updateParameter.get
126         pointsLabel.setText(score + " " + Vocabulary.POINTS)
127       })
128 
129     case TimeExpiredUpdate =>
130       disableButton(answerButtons)
