91   override def updateUI[T](update: ViewUpdate[Any]): Unit = update match
92     case DefaultUpdate => {}
93     case CurrentGameUpdate(updateParameter) =>
94       Platform.runLater(() => {
95         val gameStage: GameStage = updateParameter.get
96         courseLabel.setText(gameStage.quizInGame.course.courseId.courseName)
97         quizNumberLabel.setText(s"${gameStage.review.numberQuizAnswered+1}/${gameStage.gameSettings.asInstanceOf[StandardGameSettings].maxQuizzes}")
98         pointsLabel.setText(gameStage.quizInGame.quiz.maxScore.toString + " punti")
99         quizLabel.setText(gameStage.quizInGame.quiz.question)
100         firstAnswerButton.setText(gameStage.quizInGame.answers(0).text)
101         secondAnswerButton.setText(gameStage.quizInGame.answers(1).text)
102         thirdAnswerButton.setText(gameStage.quizInGame.answers(2).text)
103         fourthAnswerButton.setText(gameStage.quizInGame.answers(3).text)
104       })
105     case AnswerFeedbackUpdate(updateParameter) =>
106       val feedback = updateParameter.get
107       feedback._1 match
108         case 0 => if feedback._2 then firstAnswerButton.getStyleClass.add("correct-answer") else firstAnswerButton.getStyleClass.add("wrong-answer")
109         case 1 => if feedback._2 then secondAnswerButton.getStyleClass.add("correct-answer") else secondAnswerButton.getStyleClass.add("wrong-answer")
110         case 2 => if feedback._2 then thirdAnswerButton.getStyleClass.add("correct-answer") else thirdAnswerButton.getStyleClass.add("wrong-answer")
111         case 3 => if feedback._2 then fourthAnswerButton.getStyleClass.add("correct-answer") else fourthAnswerButton.getStyleClass.add("wrong-answer")
112     case TimerUpdate(updateParameter) =>
113       val timer: Timer = updateParameter.get
114       Platform.runLater(() => {
115         timeRemainingLabel.setText(s"${timer.getRemainingTime.toInt}")
116         timeProgressBar.setProgress(timer.getCompletionPercentage)
117       })
118     case TimeExpiredUpdate =>
119       disableButton(answerButtons)
