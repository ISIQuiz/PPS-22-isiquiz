74   def fillAnswerBox(quizAnsweredList: List[QuizAnswered]): Unit =

89         );

91         quizVBox.getChildren.addAll(quizDescLabel)
92         for
93           answerQuiz <- quizAnswered.quizInGame.answers;
94           if (showAllAnswers || answerQuiz.isCorrect || (if quizAnswered.answerPlayer.isDefined then quizAnswered.answerPlayer.get == answerQuiz else false))
95         do
96           val quizAnswerLabel = Label(s"${answerQuiz.text}")
97           if answerQuiz.isCorrect then quizAnswerLabel.getStyleClass.setAll("review-correct-answer") else quizAnswerLabel.getStyleClass.setAll("review-wrong-answer")
98           quizVBox.getChildren.addAll(quizAnswerLabel);
99         quizAnsweredVBox.setSpacing(10);
100         quizAnsweredVBox.getChildren.addAll(quizVBox);
101     })
