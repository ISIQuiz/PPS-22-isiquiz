100   override def updateUI[T](update: ViewUpdate[Any]): Unit = update match
101     case DefaultUpdate =>
102       Platform.runLater { () =>
103         val p = AppController.session.playerStats
104         globalTotalScoreLabel.setText(p.totalScore.toString)
105         globalTotalAnsweredQuestionsLabel.setText(p.totalAnsweredQuestions.toString)
106         globalTotalCorrectAnswersLabel.setText(p.totalCorrectAnswers.toString)
107         globalTotalAnswerPrecisionLabel.setText(p.totalAnswerPrecision.toString + " %")
108         globalTotalAverageTimeAnswerLabel.setText(p.totalAverageTimeAnswer.toString + " s")
109       }
110     case CourseInStatsListUpdate(updateParameter) =>
111       Platform.runLater { () =>
112         coursesVBox.getChildren.clear()
113         updateParameter.get.foreach(courseInStats =>
114           val radioButton = RadioButton(s"${courseInStats.course.courseId.courseName} (${courseInStats.quizInStatsList.size} quiz)");
115           radioButton.setToggleGroup(toggleCourseInStatsGroup);
116           radioButton.getStyleClass.add("label-dark");
117           radioButton.addEventHandler(MouseEvent.MOUSE_PRESSED, event =>
118             val playerStats = updatePlayerStats(List(courseInStats))
119             setTextLabels(
120               courseInStats.course.courseId.courseName,
121               playerStats.totalScore,
122               playerStats.totalAnsweredQuestions,
123               playerStats.totalCorrectAnswers,
124               playerStats.totalAnswerPrecision,
125               playerStats.totalAverageTimeAnswer
126             )
127             import controller.StatisticsMenuController.SelectCourseAction
128             sendEvent(SelectCourseAction(Option(courseInStats)));
129           );
130           coursesVBox.getChildren.addAll(radioButton)
131         )
132       }
133     case QuizInStatsListUpdate(updateParameter) =>
134       Platform.runLater { () =>
135         quizVBox.getChildren.clear()
136         updateParameter.get.foreach(quizInStats =>
137           val question = QuizInStats.getQuizQuestionById(quizInStats.quizId, AppController.session.savedCourses)
138           val radioButton = RadioButton(question);
139           radioButton.setToggleGroup(toggleQuizGroup);
140           radioButton.getStyleClass.add("label-dark");
141           radioButton.addEventHandler(MouseEvent.MOUSE_PRESSED, event =>
142             setTextLabels(
143               question,
144               quizInStats.totalScore,
145               quizInStats.totalSeen,
146               quizInStats.totalRightAnswers,
147               calculateTotalAnswerPrecision(quizInStats.totalSeen, quizInStats.totalRightAnswers),
148               quizInStats.averageTimeAnswer
149             )
150             import controller.StatisticsMenuController.SelectQuizAction
151             sendEvent(SelectQuizAction(Option(quizInStats)));
152           );
153           quizVBox.getChildren.addAll(radioButton)
154         )
155       }
156     case _ => {}
