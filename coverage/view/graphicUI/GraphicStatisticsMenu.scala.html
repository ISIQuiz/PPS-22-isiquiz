160   override def updateUI[T](update: ViewUpdate[Any]): Unit = update match
161     case DefaultUpdate =>
162       Platform.runLater { () =>
163         val p = AppController.session.playerStats
164         globalTotalScoreLabel.setText(p.totalScore.toString)
165         globalTotalAnsweredQuestionsLabel.setText(p.totalAnsweredQuestions.toString)
166         globalTotalCorrectAnswersLabel.setText(p.totalCorrectAnswers.toString)
167         globalTotalAnswerPrecisionLabel.setText(p.totalAnswerPrecision.toString + " %")
168         globalTotalAverageTimeAnswerLabel.setText(p.totalAverageTimeAnswer.toString + " s")
169       }
170     case CourseInStatsListUpdate(updateParameter) =>
171       Platform.runLater { () =>
172         coursesVBox.getChildren.clear()
173         updateParameter.get.foreach(courseInStats =>
174           val radioButton = RadioButton(s"${courseInStats.course.courseId.courseName} (${courseInStats.quizInStatsList.size} quiz)");
175           radioButton.setToggleGroup(toggleCourseInStatsGroup);
176           radioButton.getStyleClass.add("label-dark");
177           radioButton.addEventHandler(MouseEvent.MOUSE_PRESSED, event =>
178             radioButton.setSelected(true);
179             val playerStats = updatePlayerStats(List(courseInStats))
180             setTextLabels(
181               courseInStats.course.courseId.courseName,
182               playerStats.totalScore,
183               playerStats.totalAnsweredQuestions,
184               playerStats.totalCorrectAnswers,
185               playerStats.totalAnswerPrecision,
186               playerStats.totalAverageTimeAnswer
187             )
188             import controller.StatisticsMenuController.SelectCourseAction
189             sendEvent(SelectCourseAction(Option(courseInStats)));
190           );
191           coursesVBox.getChildren.addAll(radioButton)
192         )
193       }
194     case QuizInStatsListUpdate(updateParameter) =>
195       Platform.runLater { () =>
196         quizVBox.getChildren.clear()
197         updateParameter.get.foreach(quizInStats =>
198           val question = QuizInStats.getQuizQuestionById(quizInStats.quizId, AppController.session.savedCourses)
199           val radioButton = RadioButton(question);
200           radioButton.setToggleGroup(toggleQuizGroup);
201           radioButton.getStyleClass.add("label-dark");
202           radioButton.addEventHandler(MouseEvent.MOUSE_PRESSED, event =>
203             radioButton.setSelected(true);
204             setTextLabels(
205               question,
206               quizInStats.totalScore,
207               quizInStats.totalSeen,
208               quizInStats.totalRightAnswers,
209               calculateTotalAnswerPrecision(quizInStats.totalSeen, quizInStats.totalRightAnswers),
210               quizInStats.averageTimeAnswer
211             )
212             import controller.StatisticsMenuController.SelectQuizAction
213             sendEvent(SelectQuizAction(Option(quizInStats)));
214           );
215           quizVBox.getChildren.addAll(radioButton)
216         )
217       }
218     case _ => {}
