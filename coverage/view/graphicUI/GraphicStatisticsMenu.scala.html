99   override def updateUI[T](update: ViewUpdate[Any]): Unit = update match
100     case DefaultUpdate =>
101       Platform.runLater { () =>
102         val p = AppController.session.playerStats
103         globalTotalScoreLabel.setText(p.totalScore.toString)
104         globalTotalAnsweredQuestionsLabel.setText(p.totalAnsweredQuestions.toString)
105         globalTotalCorrectAnswersLabel.setText(p.totalCorrectAnswers.toString)
106         globalTotalAnswerPrecisionLabel.setText(p.totalAnswerPrecision.toString + " %")
107         globalTotalAverageTimeAnswerLabel.setText(p.totalAverageTimeAnswer.toString + " s")
108       }
109     case CourseInStatsListUpdate(updateParameter) =>
110       Platform.runLater { () =>
111         coursesVBox.getChildren.clear()
112         updateParameter.get.foreach(courseInStats =>
113           val radioButton = RadioButton(s"${courseInStats.course.courseId.courseName} (${courseInStats.quizInStatsList.size} quiz)");
114           radioButton.setToggleGroup(toggleCourseInStatsGroup);
115           radioButton.getStyleClass.add("label-dark");
116           radioButton.addEventHandler(MouseEvent.MOUSE_PRESSED, event =>
117             val playerStats = updatePlayerStats(List(courseInStats))
118             setTextLabels(
119               courseInStats.course.courseId.courseName,
120               playerStats.totalScore,
121               playerStats.totalAnsweredQuestions,
122               playerStats.totalCorrectAnswers,
123               playerStats.totalAnswerPrecision,
124               playerStats.totalAverageTimeAnswer
125             )
126             import controller.StatisticsMenuController.SelectCourseAction
127             sendEvent(SelectCourseAction(Option(courseInStats)));
128           );
129           coursesVBox.getChildren.addAll(radioButton)
130         )
131       }
132     case QuizInStatsListUpdate(updateParameter) =>
133       Platform.runLater { () =>
134         quizVBox.getChildren.clear()
135         updateParameter.get.foreach(quizInStats =>
136           val question = QuizInStats.getQuizQuestionById(quizInStats.quizId, AppController.session.savedCourses)
137           val radioButton = RadioButton(question);
138           radioButton.setToggleGroup(toggleQuizGroup);
139           radioButton.getStyleClass.add("label-dark");
140           radioButton.addEventHandler(MouseEvent.MOUSE_PRESSED, event =>
141             setTextLabels(
142               question,
143               quizInStats.totalScore,
144               quizInStats.totalSeen,
145               quizInStats.totalRightAnswers,
146               calculateTotalAnswerPrecision(quizInStats.totalSeen, quizInStats.totalRightAnswers),
147               quizInStats.averageTimeAnswer
148             )
149             import controller.StatisticsMenuController.SelectQuizAction
150             sendEvent(SelectQuizAction(Option(quizInStats)));
151           );
152           quizVBox.getChildren.addAll(radioButton)
153         )
154       }
155     case _ => {}
